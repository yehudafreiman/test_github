git checkout -b dockerfile
git push -u origin dockerfile
touch DOCKERFILE.md
mkdir streamlit_minikube_app
cd streamlit_minikube_app
touch Dockerfile main.py requirements.txt
העתקת תוכן אפליקציה לקבצים הרלוונטיים
cd ..
git add streamlit_minikube_app DOCKERFILE.md
git commit -m "Implement Dockerfile and app files"
git push origin dockerfile
cd streamlit_minikube_app

בנייה
docker build -t yehudafreiman/streamlit:v1 .
docker run -d -p 8090:8080 yehudafreiman/streamlit:v1
docker ps
docker push yehudafreiman/streamlit:v1
cd ..
